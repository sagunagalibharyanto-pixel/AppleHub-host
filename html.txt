function renderProducts(){
let list=[...products];
let q=search.value.toLowerCase();
let s=sort.value;
list=list.filter(p=>p.name.toLowerCase().includes(q));
if(s==="murah")list.sort((a,b)=>a.price-b.price);
if(s==="mahal")list.sort((a,b)=>b.price-a.price);
if(s==="rating")list.sort((a,b)=>b.r-a.r);

productList.innerHTML="";
list.forEach(p=>{
productList.innerHTML+=`
<div class="product">
<img src="${p.image}" alt="${p.name}" onerror="this.src='https://via.placeholder.com/80x80?text=No+Image'">
<div class="product-details">
<h4>${p.name}</h4>
<div class="rating">â˜… ${p.r}</div>
<div class="price">Rp ${p.price.toLocaleString()}</div>
<button onclick="addCart('${p.name}')">Tambah ke Keranjang</button>
</div>
</div>`;
});
}

function addCart(n){
let p=products.find(x=>x.name===n);
let f=cart.find(c=>c.name===n);
if(f)f.qty++;else cart.push({name:p.name,price:p.price,qty:1});
updateCart();
}

function updateCart(){
cartItems.innerHTML="";
let total=0;
if(cart.length===0){cartItems.innerHTML="Kosong";totalHarga.innerHTML="";return;}
cart.forEach((c,i)=>{
total+=c.price*c.qty;
cartItems.innerHTML+=`
<div class="cart-item">
<span>${c.name}</span>
<span class="qty">
<button onclick="cart[${i}].qty--;updateCart()">-</button>
${c.qty}
<button onclick="cart[${i}].qty++;updateCart()">+</button>
</span>
</div>`;
});
cart=cart.filter(c=>c.qty>0);
totalHarga.innerHTML="Rp "+total.toLocaleString();
}
